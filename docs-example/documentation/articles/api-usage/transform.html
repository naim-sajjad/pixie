Transform

Transform tool allows flipping and vertical/horizontal rotation/skewing of photo.

var pixie = new Pixie({
    onLoad: function() {
        var transformTool = pixie.getTool('transform');
        transformTool.rotate('basic', 90, 'left');
    }
});
